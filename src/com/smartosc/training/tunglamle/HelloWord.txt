public List<Book> getAllBook(){
        Connection conn = null;
        CallableStatement callSt = null;
        List<Book> listBook = null;
        try {
            // khoi tao doi tuong conn de lam viec voi Db
            conn = ConnectionDB.openConnection();
            // khoi tao doi tuong callST de goi Proc
            callSt = conn.prepareCall("{call getAllBook}");
            // Thuc hien goi proc
            ResultSet rs = callSt.executeQuery();
            listBook = new ArrayList<>();
            while(rs.next()){
                Book b = new Book();
                b.setBookId(rs.getInt("BookId"));
                b.setBookName(rs.getString("BookName"));
                b.setPrice(rs.getFloat("Price"));
                b.setTitle(rs.getString("Title"));
                b.setBookDescription(rs.getString("BookDescription"));
                b.setAuthor(rs.getString("Author"));
                b.setStatus(rs.getBoolean("BookStatus"));
                listBook.add(b);
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
        return listBook;
    }
